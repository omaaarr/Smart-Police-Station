/* Table Crime */
Create table Crime
  (id number constraint pk_id primary key NOT NULL,
   dateCrime date,
   lieu varchar2(50),
   categorie varchar2(50),   
   description varchar2(200)
   
 );

Create table citoyen(
   cin varchar2(10) constraint pk_cin primary key NOT NULL,
   nom varchar2(50),
   prenom varchar2(50),
   adresse varchar2(100),
   telephone number,
   email varchar2(50)
  );
